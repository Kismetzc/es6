## 响应式函数的封装



```js
// 封装一个响应式函数
let reactiveFns = []

function watchFn(fn) {
    reactiveFns.push(fn)
}


// 对象的响应式
const obj = {
    name: "why",
    age: 18
}

watchFn(function() {
    const newName = obj.name
    console.log("2323");
    console.log("fewfw");
    console.log(obj.name);
})

function bar() {
    console.log("普通函数");
}

obj.name = "kobe"
reactiveFns.forEach(fn => {
    fn()
})
```
